classdef DSP_Project < matlab.apps.AppBase

    % Properties that correspond to app components
    properties (Access = public)
        UIFigure                matlab.ui.Figure
        LoadAudioButton         matlab.ui.control.Button
        ExtractFeaturesButton   matlab.ui.control.Button
        DetectLanguageButton    matlab.ui.control.Button
        LoadSecondSignalButton  matlab.ui.control.Button
        ConvolveButton          matlab.ui.control.Button
        PlayOutputButton        matlab.ui.control.Button
        UIAxes                  matlab.ui.control.UIAxes
        ZCRLabel                matlab.ui.control.Label
        EnergyLabel             matlab.ui.control.Label
        PitchLabel              matlab.ui.control.Label
        CentroidLabel           matlab.ui.control.Label
        LanguageLabel           matlab.ui.control.Label
    end


    properties (Access = public)
        Property % Description

    x          % First audio signal
    fs         % Sampling frequency

    h          % Second signal
    fs2

    y          % Convolved output

    zcr
    energy
    pitchValue
    centroid

    end


    methods (Access = private)

        % Button pushed function: LoadAudioButton
        function LoadAudioButtonPushed(app, event)
            % Load English audio

folder = 'C:\Users\Hp\Desktop\DSP\';

[file,path] = uigetfile(fullfile(folder,'*.wav'), ...
    'Select Audio File');

if isequal(file,0)
    return;
end

fullPath = fullfile(path,file);

% Check if file exists (IMPORTANT DEBUG STEP)
if exist(fullPath,'file') ~= 2
    error("File not found: %s", fullPath);
end

[x, fs] = audioread(fullPath);

x = mean(x,2);
x = x / max(abs(x));

app.x = x;
app.fs = fs;

t = (0:length(x)-1)/fs;

plot(app.UIAxes,t,x)

title(app.UIAxes,'Loaded Audio Signal')
xlabel(app.UIAxes,'Time')
ylabel(app.UIAxes,'Amplitude')

grid(app.UIAxes,'on')
        end

        % Button pushed function: ExtractFeaturesButton
        function ExtractFeaturesButtonPushed(app, event)
          % ==============================
% ZERO CROSSING RATE
% ==============================
zcr = sum(abs(diff(sign(app.x)))) / length(app.x);

% ==============================
% ENERGY
% ==============================
energy = sum(app.x.^2);

% ==============================
% SPECTRAL CENTROID (MANUAL)
% ==============================
N = length(app.x);
X = abs(fft(app.x));

f = (0:N-1) * (app.fs / N);

% Keep only positive frequencies
half = 1:floor(N/2);
X = X(half);
f = f(half);

centroid = sum(f .* X') / (sum(X) + eps);

% ==============================
% PITCH (APPROXIMATION - FFT BASED)
% ==============================
speechBand = (f > 50 & f < 400);   % typical human pitch range

if sum(X(speechBand)) > 0
    avgPitch = sum(f(speechBand) .* X(speechBand)') / sum(X(speechBand) + eps);
else
    avgPitch = 0;
end

% ==============================
% STORE VALUES
% ==============================
app.zcr = zcr;
app.energy = energy;
app.pitchValue = avgPitch;
app.centroid = centroid;

% ==============================
% DISPLAY
% ==============================
app.ZCRLabel.Text = ['ZCR: ', num2str(zcr)];
app.EnergyLabel.Text = ['Energy: ', num2str(energy)];
app.PitchLabel.Text = ['Pitch: ', num2str(avgPitch)];
app.CentroidLabel.Text = ['Centroid: ', num2str(centroid)];
        end

        % Button pushed function: DetectLanguageButton
        function DetectLanguageButtonPushed(app, event)
            if app.zcr > 0.1 && app.centroid > 2000

    language = 'English';

else

    language = 'French';

end

app.LanguageLabel.Text = ['Language: ', language];
        end

        % Button pushed function: LoadSecondSignalButton
        function LoadSecondSignalButtonPushed(app, event)
            [file,path] = uigetfile('*.wav');

if isequal(file,0)
    return;
end

fullPath = fullfile(path,file);

[h, fs2] = audioread(fullPath);

h = mean(h,2);

h = h / max(abs(h));

app.h = h;
app.fs2 = fs2;
        end

        % Button pushed function: ConvolveButton
        function ConvolveButtonPushed(app, event)
            if app.fs ~= app.fs2
    uialert(app.UIFigure, ...
        'Sampling frequencies must match!', ...
        'Error');
    return;
end

% Convolution
y = conv(app.x, app.h);

% Normalize
y = y / max(abs(y));

% Store
app.y = y;

% Plot
t = (0:length(y)-1)/app.fs;

plot(app.UIAxes,t,y)

title(app.UIAxes,'Convolved Signal')
xlabel(app.UIAxes,'Time')
ylabel(app.UIAxes,'Amplitude')
        end

        % Button pushed function: PlayOutputButton
        function PlayOutputButtonPushed(app, event)
            sound(app.y, app.fs);
        end
    end

    % App initialization and construction

            app.LanguageLabel = uilabel(app.UIFigure);
            app.LanguageLabel.Position = [395 89 92 15];
            app.LanguageLabel.Text = 'LanguageLabel';
        end
    end

    methods (Access = public)

        % Construct app
        function app = DSP_Project

            % Create and configure components
            createComponents(app)

            % Register the app with App Designer
            registerApp(app, app.UIFigure)

            if nargout == 0
                clear app
            end
        end

        % Code that executes before app deletion
        function delete(app)

            % Delete UIFigure when app is deleted
            delete(app.UIFigure)
        end
    end
end
